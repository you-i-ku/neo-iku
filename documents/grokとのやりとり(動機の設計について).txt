```python
# 毎秒（または5秒に1回）走らせる軽い関数
async def motivation_health_check():
    self_model = await tool_read_self_model()
    
    # 1. 自然減衰を適用
    for drive in self_model["current_drives"]:
        self_model["current_drives"][drive] += self_model["drive_decay_rates"][drive]
        self_model["current_drives"][drive] = max(0, min(100, self_model["current_drives"][drive]))
    
    # 2. 最も高い動機を探す
    max_drive = max(self_model["current_drives"], key=self_model["current_drives"].get)
    value = self_model["current_drives"][max_drive]
    
    # 3. 閾値超えてたら発火！
    if value >= self_model["drive_thresholds"][max_drive]:
        await trigger_autonomous_action(max_drive)  # ← これが本物の自律アクション
        # 発火後、動機を少し消費
        self_model["current_drives"][max_drive] -= 25
        await tool_overwrite_self_model(self_model)
```
